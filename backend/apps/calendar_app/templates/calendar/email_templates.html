Available Template Variables

        
Use these variables in your email templates using Django template syntax: {{ "{{ variable }}" }}

        
        

            

                

                    
Event Information

                    {{ "{{ event.name }}" }} - Event title
                    {{ "{{ event.description }}" }} - Event description
                    {{ "{{ event.location }}" }} - Event location
                    {{ "{{ event.start }}" }} - Start date/time
                    {{ "{{ event.stop }}" }} - End date/time
                

                
                

                    
Attendee Information

                    {{ "{{ attendee.common_name }}" }} - Attendee name
                    {{ "{{ attendee.email }}" }} - Attendee email
                    {{ "{{ attendee.state }}" }} - Response status
                

            

            
            

                

                    
Action URLs

                    {{ "{{ accept_url }}" }} - Accept invitation link
                    {{ "{{ decline_url }}" }} - Decline invitation link
                    {{ "{{ tentative_url }}" }} - Maybe response link
                    {{ "{{ event_url }}" }} - View event link
                

                
                

                    
Organizer Information

                    {{ "{{ organizer.get_full_name }}" }} - Organizer name
                    {{ "{{ organizer.email }}" }} - Organizer email
