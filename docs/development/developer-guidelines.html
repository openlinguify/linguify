Code Style Standards

Python Code Style

We follow PEP 8 with some project-specific conventions:

# models.py
from django.db import models
from django.contrib.auth import get_user_model

User = get_user_model()


class Course(models.Model):
    """Educational course model with progress tracking."""
    
    title = models.CharField(max_length=200)
    description = models.TextField(blank=True)
    created_by = models.ForeignKey(
        User, 
        on_delete=models.CASCADE,
        related_name='created_courses'
    )
    created_at = models.DateTimeField(auto_now_add=True)
    updated_at = models.DateTimeField(auto_now=True)
    
    class Meta:
        ordering = ['-created_at']
        verbose_name_plural = 'courses'
    
    def __str__(self):
        return self.title
    
    def get_absolute_url(self):
        """Return the canonical URL for this course."""
        return reverse('course:detail', kwargs={'pk': self.pk})

Key Conventions

  • Line length: Maximum 88 characters (Black formatter)
  • Imports: Use isort for consistent import ordering
  • Docstrings: Use triple quotes for all classes and functions
  • Naming: snake_case for variables and functions, PascalCase for classes
  • Type hints: Use where appropriate, especially for public APIs

Django Specific Guidelines

Model Guidelines

Best Practices
  • Always add created_at and updated_at timestamps
  • Use descriptive related_name for foreign keys
  • Include proper __str__ methods
  • Add get_absolute_url for models with detail views
  • Use blank=True for optional fields, null=True sparingly

View Guidelines

Class-Based View Example
# views.py
from django.views.generic import DetailView
from django.contrib.auth.mixins import LoginRequiredMixin
from django.shortcuts import get_object_or_404

from .models import Course


class CourseDetailView(LoginRequiredMixin, DetailView):
    """Display course details with user progress."""
    
    model = Course
    template_name = 'course/course_detail.html'
    context_object_name = 'course'
    
    def get_context_data(self, **kwargs):
        context = super().get_context_data(**kwargs)
        context['user_progress'] = self.get_user_progress()
        context['lessons'] = self.object.lessons.all()
        return context
    
    def get_user_progress(self):
        """Get current user's progress for this course."""
        if self.request.user.is_authenticated:
            return UserProgress.objects.filter(
                user=self.request.user,
                course=self.object
            ).first()
        return None

Development Workflow

Git Workflow

We use a feature branch workflow with the following conventions:

Branch Naming

  • feature/add-lesson-progress - New features
  • fix/course-enrollment-bug - Bug fixes
  • refactor/user-progress-model - Code refactoring
  • docs/api-documentation - Documentation updates
  • translations/add-german - Translation work

Commit Messages

Commit Message Format
# Good commit messages
git commit -m "Add user progress tracking to course model

- Added UserProgress model with completion percentage
- Implemented progress calculation for lessons
- Added API endpoints for progress updates"

git commit -m "Fix course enrollment validation bug

- Prevent duplicate enrollments for same user/course
- Add proper error handling in enrollment view
- Update tests for enrollment edge cases"
Commit Message Tips
  • Use imperative mood ("Add feature" not "Added feature")
  • First line should be 50 characters or less
  • Include detailed description for complex changes
  • Reference issue numbers when applicable

Pull Request Process

  1. Create feature branch: git checkout -b feature/your-feature
  2. Make changes and commit following our guidelines
  3. Write/update tests for your changes
  4. Run the test suite: pytest
  5. Format code: black . && isort .
  6. Push branch: git push origin feature/your-feature
  7. Create pull request with detailed description

Documentation Standards

Code Documentation

Docstring Example
def calculate_course_progress(user, course):
    """
    Calculate user's progress percentage for a given course.
    
    Args:
        user (User): The user to calculate progress for
        course (Course): The course to check progress against
        
    Returns:
        float: Progress percentage between 0.0 and 100.0
        
    Raises:
        ValueError: If user is not enrolled in the course
        
    Example:
        >>> user = User.objects.get(email='test@example.com')
        >>> course = Course.objects.get(title='Python Basics')
        >>> progress = calculate_course_progress(user, course)
        >>> assert 0.0 <= progress <= 100.0
    """
    if not course.enrollments.filter(user=user).exists():
        raise ValueError("User is not enrolled in this course")
    
    completed_lessons = user.completed_lessons.filter(course=course).count()
    total_lessons = course.lessons.count()
    
    if total_lessons == 0:
        return 0.0
        
    return (completed_lessons / total_lessons) * 100

Data Validation

Form Validation Example
# forms.py
from django import forms
from django.core.exceptions import ValidationError

class CourseCreationForm(forms.ModelForm):
    """Form for creating new courses with validation."""
    
    class Meta:
        model = Course
        fields = ['title', 'description', 'language', 'difficulty']
    
    def clean_title(self):
        """Validate course title is unique for this user."""
        title = self.cleaned_data['title']
        user = self.instance.created_by if self.instance.pk else None
        
        if Course.objects.filter(title=title, created_by=user).exists():
            raise ValidationError(
                "You already have a course with this title."
            )
        return title
    
    def clean(self):
        """Cross-field validation."""
        cleaned_data = super().clean()
        title = cleaned_data.get('title')
        description = cleaned_data.get('description')
        
        if title and description and title.lower() in description.lower():
            raise ValidationError(
                "Course description should not repeat the title."
            )
        
        return cleaned_data

Performance Guidelines

Database Optimization

Query Optimization
  • Use select_related() for foreign key relationships
  • Use prefetch_related() for many-to-many relationships
  • Add database indexes for frequently queried fields
  • Use only() and defer() to limit field selection
  • Implement pagination for large datasets
Optimized Query Example
# Optimized queryset for course list with related data
courses = Course.objects.select_related(
    'created_by'
).prefetch_related(
    'lessons',
    'enrollments__user'
).filter(
    is_published=True
).only(
    'title', 'description', 'created_by__username'
).order_by('-created_at')

Caching Strategy

# Use caching for expensive operations
from django.core.cache import cache

def get_popular_courses():
    """Get popular courses with caching."""
    cache_key = 'popular_courses_list'
    popular_courses = cache.get(cache_key)
    
    if popular_courses is None:
        popular_courses = Course.objects.filter(
            enrollments__count__gte=10
        ).select_related('created_by').order_by('-enrollment_count')[:10]
        
        cache.set(cache_key, popular_courses, timeout=3600)  # 1 hour
    
    return popular_courses

Translation Best Practices

Python Translation Example
from django.utils.translation import gettext_lazy as _
from django.utils.translation import ngettext

class Course(models.Model):
    title = models.CharField(_("Course title"), max_length=200)
    description = models.TextField(_("Course description"), blank=True)
    
    class Meta:
        verbose_name = _("Course")
        verbose_name_plural = _("Courses")

# For messages with pluralization
def format_lesson_count(count):
    return ngettext(
        "%(count)d lesson",
        "%(count)d lessons", 
        count
    ) % {'count': count}
